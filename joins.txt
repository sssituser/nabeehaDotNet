joins

1.Left outer join
2.Right outer join
3.Fullouter join
4.Inner join(self join)
5.Cross joins

create table fyear(stu_id int primary key,stu_name varchar(10) not null)
create table Syear(stu_id int primary key,stu_name varchar(10) not null)
insert into fyear values(11,'kiran'),(12,'Ravi'),(13,'shiva'),(14,'Raj')
insert into syear values(15,'kumar')
select * from fyear
select * from Syear
select f.stu_id,s.stu_name from fyear f inner join  syear s on f.stu_id =s.stu_id
select * from fyear f left join  syear s on f.stu_id =s.stu_id
select * from fyear f right join  syear s on f.stu_id =s.stu_id
select * from fyear cross join syear
